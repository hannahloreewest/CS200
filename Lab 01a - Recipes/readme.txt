#include <iostream>     // Input/output library
using namespace std;    // Standard C++ library namespace

int main()
{
	// Program begins
	double baking_soda = 1;
	double baking_powder = .5;
	double butter = 1;
	double white_sugar = 1.5;
	double egg = 1;
	double ratio;

	cout << "How many batches would you like to make? ";
    cin >> ratio;

	cout << (baking_soda * ratio) << " tsp baking soda" << endl;
	cout <<  (baking_powder * ratio) << " tsp baking powder" << endl;
	cout << (butter * ratio) << " c butter" << endl;
	cout << (white_sugar * ratio) << " c white sugar" << endl;
	cout << (egg * ratio) << " egg" << endl;
 	
	
	
	
	
	
	while (true);    // Program stops here
	return 0;
}
